105  void rightMultiply(double*& dest, const double* src) const
106  {
107  int destSize=cols();
108 
109  if (! dest){
110  dest=new double [ destSize ];
111  memset(dest,0, destSize*sizeof(double));
112  }
113 
114  // map the memory by Eigen
115  Eigen::Map<Eigen::VectorXd> destVec(dest, destSize);
116  Eigen::Map<const Eigen::VectorXd> srcVec(src, rows());
117 
118 # ifdef G2O_OPENMP
119 # pragma omp parallel for default (shared) schedule(dynamic, 10)
120 # endif
121  for (int i=0; i < static_cast<int>(_blockCols.size()); ++i){
122  int destOffset = colBaseOfBlock(i);
123  for (typename SparseColumn::const_iterator it = _blockCols[i].begin(); it!=_blockCols[i].end(); ++it) {
124  const SparseMatrixBlock* a = it->block;

146  int fillCCS(int* Cp, int* Ci, double* Cx, bool upperTriangle = false) const
147  {
148  assert(Cp && Ci && Cx && "Target destination is NULL");
149  int nz=0;
150  for (size_t i=0; i<_blockCols.size(); ++i){
151  int cstart=i ? _colBlockIndices[i-1] : 0;
152  int csize=colsOfBlock(i);
153  for (int c=0; c<csize; ++c) {
154  *Cp=nz;
155  for (typename SparseColumn::const_iterator it = _blockCols[i].begin(); it!=_blockCols[i].end(); ++it) {
156  const SparseMatrixBlock* b=it->block;
157  int rstart=it->row ? _rowBlockIndices[it->row-1] : 0;
158 
159  int elemsToCopy = b->rows();
160  if (upperTriangle && rstart == cstart)
161  elemsToCopy = c + 1;
162  for (int r=0; r<elemsToCopy; ++r){
163  *Cx++ = (*b)(r,c);
164  *Ci++ = rstart++;
165  ++nz;
166  }
167  }
168  ++Cp;
169  }
170  }
171  *Cp=nz;
172  return nz;
173  }

179  int fillCCS(double* Cx, bool upperTriangle = false) const
180  {
181  assert(Cx && "Target destination is NULL");
182  double* CxStart = Cx;
183  int cstart = 0;
184  for (size_t i=0; i<_blockCols.size(); ++i){
185  int csize = _colBlockIndices[i] - cstart;
186  for (int c=0; c<csize; ++c) {
187  for (typename SparseColumn::const_iterator it = _blockCols[i].begin(); it!=_blockCols[i].end(); ++it) {
188  const SparseMatrixBlock* b = it->block;
189  int rstart = it->row ? _rowBlockIndices[it->row-1] : 0;
190 
191  int elemsToCopy = b->rows();
192  if (upperTriangle && rstart == cstart)
193  elemsToCopy = c + 1;
194  memcpy(Cx, b->data() + c*b->rows(), elemsToCopy * sizeof(double));
195  Cx += elemsToCopy;
196 
197  }
198  }
199  cstart = _colBlockIndices[i];
200  }
201  return Cx - CxStart;
202  }

262  assert(c <(int)_blockCols.size() && "accessing column which is not available");
263  SparseColumn& sparseColumn = _blockCols[c];
264  typename SparseColumn::iterator foundIt = sparseColumn.find(r);
265  if (foundIt == sparseColumn.end()) {
266  int rb = rowsOfBlock(r);
267  int cb = colsOfBlock(c);

271  sparseColumn[r] = m;

273  }

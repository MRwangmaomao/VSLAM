199  template < class MatrixResultType, class MatrixFactorType >
200  bool SparseBlockMatrix<MatrixType>::multiply(SparseBlockMatrix<MatrixResultType>*& dest, const SparseBlockMatrix<MatrixFactorType> * M) const {
201  // sanity check
202  if (_colBlockIndices.size()!=M->_rowBlockIndices.size())
203  return false;
204  for (size_t i=0; i<_colBlockIndices.size(); ++i){
205  if (_colBlockIndices[i]!=M->_rowBlockIndices[i])
206  return false;
207  }
208  if (! dest) {
209  dest=new SparseBlockMatrix<MatrixResultType>(&_rowBlockIndices[0],&(M->_colBlockIndices[0]), _rowBlockIndices.size(), M->_colBlockIndices.size() );
210  }
211  if (! dest->_hasStorage)
212  return false;
213  for (size_t i=0; i<M->_blockCols.size(); ++i){
214  for (typename SparseBlockMatrix<MatrixFactorType>::IntBlockMap::const_iterator it=M->_blockCols[i].begin(); it!=M->_blockCols[i].end(); ++it){
215  // look for a non-zero block in a row of column it
216  int colM=i;
217  const typename SparseBlockMatrix<MatrixFactorType>::SparseMatrixBlock *b=it->second;
218  typename SparseBlockMatrix<MatrixType>::IntBlockMap::const_iterator rbt=_blockCols[it->first].begin();
219  while(rbt!=_blockCols[it->first].end()){
220  //int colA=it->first;
221  int rowA=rbt->first;
222  typename SparseBlockMatrix<MatrixType>::SparseMatrixBlock *a=rbt->second;
223  typename SparseBlockMatrix<MatrixResultType>::SparseMatrixBlock *c=dest->block(rowA,colM,true);
224  assert (c->rows()==a->rows());
225  assert (c->cols()==b->cols());
226  ++rbt;
227  (*c)+=(*a)*(*b);
228  }
229  }
230  }
231  return false;
232  }

331  SparseBlockMatrix<MatrixType>* SparseBlockMatrix<MatrixType>::slice(int rmin, int rmax, int cmin, int cmax, bool alloc) const {
332  int m=rmax-rmin;
333  int n=cmax-cmin;
334  int rowIdx [m];
335  rowIdx[0] = rowsOfBlock(rmin);
336  for (int i=1; i<m; ++i){
337  rowIdx[i]=rowIdx[i-1]+rowsOfBlock(rmin+i);
338  }
339 
340  int colIdx [n];
341  colIdx[0] = colsOfBlock(cmin);
342  for (int i=1; i<n; ++i){
343  colIdx[i]=colIdx[i-1]+colsOfBlock(cmin+i);
344  }
345  typename SparseBlockMatrix<MatrixType>::SparseBlockMatrix* s=new SparseBlockMatrix(rowIdx, colIdx, m, n, true);
346  for (int i=0; i<n; ++i){
347  int mc=cmin+i;
348  for (typename SparseBlockMatrix<MatrixType>::IntBlockMap::const_iterator it=_blockCols[mc].begin(); it!=_blockCols[mc].end(); ++it){
349  if (it->first >= rmin && it->first < rmax){
350  typename SparseBlockMatrix<MatrixType>::SparseMatrixBlock* b = alloc ? new typename SparseBlockMatrix<MatrixType>::SparseMatrixBlock (* (it->second) ) : it->second;
351  s->_blockCols[i].insert(std::make_pair(it->first-rmin, b));
352  }
353  }
354  }
355  s->_hasStorage=alloc;
356  return s;
357  }

405  template <class MatrixType>
406  bool SparseBlockMatrix<MatrixType>::symmPermutation(SparseBlockMatrix<MatrixType>*& dest, const int* pinv, bool upperTriangle) const{
407  // compute the permuted version of the new row/column layout
408  size_t n=_rowBlockIndices.size();
409  // computed the block sizes
410  int blockSizes[_rowBlockIndices.size()];
411  blockSizes[0]=_rowBlockIndices[0];
412  for (size_t i=1; i<n; ++i){
413  blockSizes[i]=_rowBlockIndices[i]-_rowBlockIndices[i-1];
414  }
415  // permute them
416  int pBlockIndices[_rowBlockIndices.size()];
417  for (size_t i=0; i<n; ++i){
418  pBlockIndices[pinv[i]]=blockSizes[i];
419  }
420  for (size_t i=1; i<n; ++i){
421  pBlockIndices[i]+=pBlockIndices[i-1];
422  }
423  // allocate C, or check the structure;
424  if (! dest){
425  dest=new SparseBlockMatrix(pBlockIndices, pBlockIndices, n, n);
426  } else {
427  if (dest->_rowBlockIndices.size()!=n)
428  return false;
429  if (dest->_colBlockIndices.size()!=n)
430  return false;
431  for (size_t i=0; i<n; ++i){
432  if (dest->_rowBlockIndices[i]!=pBlockIndices[i])
433  return false;
434  if (dest->_colBlockIndices[i]!=pBlockIndices[i])
435  return false;
436  }
437  dest->clear();
438  }
439  // now ready to permute the columns
440  for (size_t i=0; i<n; ++i){
441  //cerr << PVAR(i) << " ";
442  int pi=pinv[i];
443  for (typename SparseBlockMatrix<MatrixType>::IntBlockMap::const_iterator it=_blockCols[i].begin();
444  it!=_blockCols[i].end(); ++it){
445  int pj=pinv[it->first];
446 
447  const typename SparseBlockMatrix<MatrixType>::SparseMatrixBlock* s=it->second;
448 
449  typename SparseBlockMatrix<MatrixType>::SparseMatrixBlock* b=0;
450  if (! upperTriangle || pj<=pi) {
451  b=dest->block(pj,pi,true);
452  assert(b->cols()==s->cols());
453  assert(b->rows()==s->rows());
454  *b=*s;
455  } else {
456  b=dest->block(pi,pj,true);
457  assert(b);
458  assert(b->rows()==s->cols());
459  assert(b->cols()==s->rows());
460  *b=s->transpose();
461  }
462  }
463  //cerr << endl;
464  // within each row,
465  }
466  return true;
467

526  void SparseBlockMatrix<MatrixType>::fillBlockStructure(MatrixStructure& ms) const
527  {
528  int n = _colBlockIndices.size();
529  int nzMax = (int)nonZeroBlocks();
530 
531  ms.alloc(n, nzMax);
532  ms.m = _rowBlockIndices.size();

535  int* Cp = ms.Ap;
536  int* Ci = ms.Aii;
537  for (int i = 0; i < static_cast<int>(_blockCols.size()); ++i){

634  {
635  // sort the sparse columns and add them to the map structures by
636  // exploiting that we are inserting a sorted structure
637  typedef std::pair<int, MatrixType*> SparseColumnPair;
638  typedef typename SparseBlockMatrixHashMap<MatrixType>::SparseColumn HashSparseColumn;
639  for (size_t i = 0; i < hashMatrix.blockCols().size(); ++i) {
640  // prepare a temporary vector for sorting
641  HashSparseColumn& column = hashMatrix.blockCols()[i];
642  if (column.size() == 0)
643  continue;
644  std::vector<SparseColumnPair> sparseRowSorted; // temporary structure
645  sparseRowSorted.reserve(column.size());
646  for (typename HashSparseColumn::const_iterator it = column.begin(); it != column.end(); ++it)
647  sparseRowSorted.push_back(*it);
648  std::sort(sparseRowSorted.begin(), sparseRowSorted.end(), CmpPairFirst<int, MatrixType*>());
649  // try to free some memory early
650  HashSparseColumn aux;
651  swap(aux, column);
652  // now insert sorted vector to the std::map structure
653  IntBlockMap& destColumnMap = blockCols()[i];
654  destColumnMap.insert(sparseRowSorted[0]);
655  for (size_t j = 1; j < sparseRowSorted.size(); ++j) {
656  typename SparseBlockMatrix<MatrixType>::IntBlockMap::iterator hint = destColumnMap.end();
657  --hint; // cppreference says the element goes after the hint (until C++11)
658  destColumnMap.insert(hint, sparseRowSorted[j]);
659  }

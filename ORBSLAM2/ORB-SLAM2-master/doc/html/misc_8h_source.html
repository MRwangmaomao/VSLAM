94 inline double normalize_theta(double theta)
95 {
96  if (theta >= -M_PI && theta < M_PI)
97  return theta;
98 
99  double multiplier = floor(theta / (2*M_PI));
100  theta = theta - multiplier*2*M_PI;
101  if (theta >= M_PI)
102  theta -= 2*M_PI;
103  if (theta < -M_PI)
104  theta += 2*M_PI;
105 
106  return theta;
107 }
